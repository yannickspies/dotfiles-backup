Owja, was even naar je login PR aan het kijken. Maar je kan je routes.txt yeeten, omdat je via provideServerRendering(withRoutes(serverRoutes)), je ServerRoutes toevoegt. Je kan dan in app.routes.server.ts onderstaande doen. Om heel verbose te maken welke routes je sowieso wilt prerenderen. Ookal heb je die wildcard catch onderaan:
 
import { ServerRoute, RenderMode } from '@angular/ssr';
export const serverRoutes: ServerRoute[] = [
  { path: '/', renderMode: RenderMode.Prerender },
  { path: '/inloggen', renderMode: RenderMode.Prerender },
  { path: '/overzicht', renderMode: RenderMode.Prerender },
  { path: '/schades', renderMode: RenderMode.Prerender },
  { path: '/mijn-gegevens', renderMode: RenderMode.Prerender },
  { path: '/mijn-facturen', renderMode: RenderMode.Prerender },
  { path: '/contact', renderMode: RenderMode.Prerender },
  { path: '/betalen', renderMode: RenderMode.Prerender },
  { path: '/betalen/ideal/starten', renderMode: RenderMode.Prerender },
  { path: '/betalen/ideal/afronden', renderMode: RenderMode.Prerender },
  { path: '**', renderMode: RenderMode.Prerender }
];
